    plt.scatter(xs, ys, marker='o', s=80, facecolors='none', edgecolors='tab:blue', linewidths=1.5)
    poly = np.polynomial.Polynomial.fit(xs, ys, deg=polyn_order, domain=(xs[0], xs[-1]))
    xf, yf = poly.linspace()
    plt.plot(xf, yf, color='tab:red')
